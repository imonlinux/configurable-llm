# Test dependencies for Configurable LLM
#
# pytest-homeassistant-custom-component tracks home-assistant/core; the patch
# number (0.13.X) increments with each HA release, and the bundled Home
# Assistant version determines most of the transitive tree (hassil,
# home-assistant-intents, aiodns, and other per-component deps).
#
# The plugin is PINNED (==) rather than a floor (>=) on purpose. An unbounded
# floor lets pip pull whichever plugin is newest at run time, bundling a newer
# HA whose transitive pins then conflict with the bare component requirements
# below — producing intermittent failures with no change to this repo. Pinning
# the plugin locks HA to a single coherent tree (HA 2025.8.1).
#
# 0.13.270 bundles homeassistant==2025.8.1, the minimum HA this integration
# targets. Plugin versions 0.13.206+ require Python >= 3.13, so CI must run on
# Python 3.13+ (see .github/workflows/tests.yaml).
pytest-homeassistant-custom-component==0.13.270

pytest>=8.0.0
pytest-asyncio>=0.23.0
pytest-cov>=4.1.0
pytest-timeout>=2.3.0

# Runtime dependency of this integration, not provided by Home Assistant itself.
anthropic==0.96.0

# Per-component requirements the base plugin does NOT install but the
# integration's platforms import at module load. Unpinned so the pinned HA tree
# decides exact versions:
#   - conversation -> `from hassil.recognize import RecognizeResult`
#   - ai_task -> camera -> `from turbojpeg import TurboJPEG`
# PyTurboJPEG's *import* succeeds without the native libturbojpeg library (only
# JPEG decoding needs it, which the tests never do), so no apt step.
hassil
home-assistant-intents
PyTurboJPEG

# Pin pycares to the 4.x line. Home Assistant 2025.8.1 pins aiodns==3.5.0, and
# aiodns 3.5.0 references `pycares.ares_query_a_result` in its type hints. That
# symbol exists in pycares 4.9.0–4.11.x but was REMOVED in pycares 5.0.0. The
# GitHub runner image pre-installs pycares 5.x, so without this pin aiodns fails
# at import with `module 'pycares' has no attribute 'ares_query_a_result'`.
# `>=4.9.0,<5` satisfies aiodns 3.5.0's own requirement (pycares>=4.9.0) and
# guarantees the symbol is present. aiodns itself is left to HA's pin (3.5.0).
pycares>=4.9.0,<5
