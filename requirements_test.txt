# Test dependencies for Configurable LLM
#
# pytest-homeassistant-custom-component tracks home-assistant/core; the patch
# number (0.13.X) increments with each HA release, and the bundled Home
# Assistant version determines the ENTIRE transitive tree (including hassil,
# home-assistant-intents, aiodns/pycares, and other per-component deps).
#
# This is PINNED (==) rather than a floor (>=) on purpose. An unbounded floor
# lets pip pull whichever plugin is newest at run time, which bundles a newer
# HA whose transitive pins then conflict with the bare component requirements
# below — producing intermittent "ModuleNotFoundError" or resolution failures
# that appear without any change to this repo. Pinning the plugin locks HA to a
# single coherent tree, so the per-component packages below resolve to exactly
# the versions that HA 2025.8.1 expects.
#
# 0.13.270 bundles homeassistant==2025.8.1, the minimum HA this integration
# targets (first tree with the ConfigSubentry / AI Task APIs it uses). Plugin
# versions 0.13.206+ require Python >= 3.13, so CI must run on Python 3.13+
# (see .github/workflows/tests.yaml). To move to a newer HA later, bump this
# single pin to the plugin version bundling the desired HA release.
#
# NOTE: the install step in the workflow uses `pip install --upgrade` so that
# pip actively resolves to Home Assistant's pinned transitive versions, rather
# than leaving newer packages that the GitHub runner image pre-installs (e.g.
# aiodns) in place. Without --upgrade, a pre-baked aiodns 4.x lingers and
# crashes at import with `pycares has no attribute ares_query_a_result`.
pytest-homeassistant-custom-component==0.13.270

pytest>=8.0.0
pytest-asyncio>=0.23.0
pytest-cov>=4.1.0
pytest-timeout>=2.3.0

# Runtime dependency of this integration, not provided by Home Assistant itself.
anthropic==0.96.0

# Per-component requirements that the base plugin does NOT install but the
# integration's platforms import at module load. Left unpinned so the pinned
# Home Assistant tree above decides the exact compatible versions:
#   - conversation -> `from hassil.recognize import RecognizeResult`
#   - ai_task -> camera -> `from turbojpeg import TurboJPEG`
# PyTurboJPEG's *import* succeeds without the native libturbojpeg system library
# (only JPEG decoding needs it, which the tests never do), so no apt step.
hassil
home-assistant-intents
PyTurboJPEG

# Keep aiodns and pycares as a MATCHED pair. aiodns is an optional aiohttp
# dependency ("speedups" extra) that the GitHub runner image pre-installs at
# 4.x. aiodns 4.x references pycares.ares_query_a_result, which only exists in
# pycares 5.x — but the runner pre-installs pycares 4.x, causing an
# AttributeError at plugin import. Rather than fight the pre-installed aiodns
# 4.x (a downgrade pin did not reliably take against the baked-in version),
# pin pycares UP to 5.x so the symbol aiodns 4.x needs is present. Both have
# cp313 wheels.
aiodns>=4.0.0
pycares>=5.0.0
